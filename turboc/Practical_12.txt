// Practical 12: Write a program to find factorial of number using recursion.




#include <iostream.h>
#include <conio.h>

long factorial(int n)
{
    if (n == 0 || n == 1)
        return 1;
    else
        return n * factorial(n - 1);
}

void main()
{
    int num;

    clrscr();

    cout << "Enter number: ";
    cin >> num;

    cout << "Factorial of " << num << " is " << factorial(num);

    getch();
}
