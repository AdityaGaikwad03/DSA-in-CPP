// Practical 5: Implement circular queue using arrays.



#include <iostream.h>
#include <conio.h>

#define MAX 20

class CircularQueue
{
    int arr[MAX];
    int front, rear, capacity;

public:
    CircularQueue(int size)
    {
        capacity = size;
        front = -1;
        rear = -1;
    }

    int isFull()
    {
        if (front == (rear + 1) % capacity)
            return 1;
        else
            return 0;
    }

    int isEmpty()
    {
        if (front == -1)
            return 1;
        else
            return 0;
    }

    void enqueue(int val)
    {
        if (isFull())
        {
            cout << "Overflow: Cannot insert " << val << endl;
            return;
        }

        if (isEmpty())
            front = rear = 0;
        else
            rear = (rear + 1) % capacity;

        arr[rear] = val;
        cout << val << " enqueued." << endl;
    }

    void dequeue()
    {
        if (isEmpty())
        {
            cout << "Underflow: No element to dequeue." << endl;
            return;
        }

        cout << arr[front] << " dequeued." << endl;

        if (front == rear)
            front = rear = -1;
        else
            front = (front + 1) % capacity;
    }

    void display()
    {
        int i;
        if (isEmpty())
        {
            cout << "Queue is empty." << endl;
            return;
        }

        cout << "Queue elements: ";
        i = front;
        while (1)
        {
            cout << arr[i] << " ";
            if (i == rear)
                break;
            i = (i + 1) % capacity;
        }
        cout << endl;
    }
};

void main()
{
    int size, choice, val;

    clrscr();

    cout << "Enter queue capacity: ";
    cin >> size;

    CircularQueue cq(size);

    do
    {
        cout << "\n1. Enqueue\n2. Dequeue\n3. Display\n4. Exit";
        cout << "\nChoice: ";
        cin >> choice;

        switch (choice)
        {
        case 1:
            cout << "Value: ";
            cin >> val;
            cq.enqueue(val);
            break;

        case 2:
            cq.dequeue();
            break;

        case 3:
            cq.display();
            break;

        case 4:
            cout << "Exit...";
            break;

        default:
            cout << "Invalid choice!";
        }

    } while (choice != 4);

    getch();
}
